**Create reporters/test/factories/rspec.ts** following the exact pattern of reporters/test/factories/pytest.ts:

```typescript
import { spawnSync } from 'node:child_process'
import { join } from 'node:path'
import type { ReporterConfig, TestScenarios } from '../types'
import { copyTestArtifacts } from './helpers'

export function createRspecReporter(): ReporterConfig {
  const artifactDir = 'rspec'
  const testScenarios = {
    singlePassing: 'single_passing_spec.rb',
    singleFailing: 'single_failing_spec.rb',
    singleImportError: 'single_import_error_spec.rb',
  }

  return {
    name: 'RspecReporter',
    testScenarios,
    run: (tempDir, scenario: keyof TestScenarios) => {
      copyTestArtifacts(artifactDir, testScenarios, scenario, tempDir)

      const rspecDir = join(__dirname, '../../rspec')
      const testFile = testScenarios[scenario]

      // Run rspec with bundler from the rspec reporter directory
      // Use --require to load the formatter and --format to activate it
      spawnSync(
        'bundle',
        [
          'exec', 'rspec', join(tempDir, testFile),
          '--format', 'TddGuardRspec::Formatter',
          '--require', join(rspecDir, 'lib/tdd_guard_rspec'),
          '--no-color',
        ],
        {
          cwd: tempDir,
          stdio: 'pipe',
          encoding: 'utf8',
          env: {
            ...process.env,
            BUNDLE_GEMFILE: join(rspecDir, 'Gemfile'),
          },
        }
      )
    },
  }
}
```

Key differences from pytest factory:
- Uses `bundle exec rspec` instead of a venv python path
- Sets BUNDLE_GEMFILE env var to point to the rspec reporter's Gemfile
- Passes `--format TddGuardRspec::Formatter` and `--require` with the full path to the lib
- cwd is tempDir (so test.json is written relative to tempDir)
- Uses `--no-color` to avoid ANSI codes in error output
